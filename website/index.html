Quick Start

Getting started with WebSkill is seamless. Install our CLI and start defining your agentic interface contracts in minutes.

# Terminal
npm install web-skill
npx webskill init
webskill.config.ts
import { createWebSkillGenerator } from 'web-skill';
import { z } from 'zod';

const webSkills = createWebSkillGenerator();

const orderSkill = webSkills.newSkill({
name: 'PrepareOrderSkill',
title: 'Prepares actions for orders',
});

orderSkill.addFunction(
(input) => store.prepareOrder(input),
'prepareOrder',
{
inputSchema: z.object({
orderId: z.string(),
priority: z.enum(['low', 'high'])
})
}
);

webSkills.install();
